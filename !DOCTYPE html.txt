    const deoriaBtn = document.getElementById("deoriaBtn");
    const backBtnDeoria = document.getElementById("backBtnDeoria");
    const popup = document.getElementById("imagePopup");
    const popupImg = document.getElementById("popupImg");
    const domainsBtn = document.getElementById("domainsBtn");
    const domainsPage = document.getElementById("domainsPage");
    const backBtnDomains = document.getElementById("backBtnDomains");

    historyBtn.addEventListener("click", () => switchPage(mainPage, historyPage));
    backBtn.addEventListener("click", e => { e.preventDefault(); switchPage(historyPage, mainPage); });
    deoriaBtn.addEventListener("click", () => switchPage(mainPage, deoriaPage));
    backBtnDeoria.addEventListener("click", e => { e.preventDefault(); switchPage(deoriaPage, mainPage); });
domainsBtn.addEventListener("click", () => switchPage(mainPage, domainsPage));
backBtnDomains.addEventListener("click", e => {
  e.preventDefault();
  switchPage(domainsPage, mainPage);
});

    function switchPage(hide, show) {
      hide.classList.remove("active");
      setTimeout(() => {
        hide.style.display = "none";
        show.style.display = "block";
        setTimeout(() => show.classList.add("active"), 50);
        window.scrollTo(0, 0);
      }, 500);
    }

    document.querySelectorAll("img").forEach(img => {
      img.addEventListener("click", () => {
        popup.style.display = "flex";
        popupImg.src = img.src;
        setTimeout(() => {
          popupImg.style.transform = "scale(1)";
          popupImg.style.opacity = "1";
        }, 50);
      });
    });

    popup.addEventListener("click", () => {
      popupImg.style.transform = "scale(0.8)";
      popupImg.style.opacity = "0";
      setTimeout(() => popup.style.display = "none", 300);
    });
  </script>
